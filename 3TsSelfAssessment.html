    <script>

        document.getElementById('assessmentForm').addEventListener('submit', function(e) {

            e.preventDefault();

           

            // Collect answers

            const q1 = document.querySelector('input[name="q1"]:checked').value;

            const q2 = document.querySelector('input[name="q2"]:checked').value;

            const q3 = document.querySelector('input[name="q3"]:checked').value;

            const q4 = document.querySelector('input[name="q4"]:checked').value;

            const q5 = document.querySelector('input[name="q5"]:checked').value;

           

            // Count responses

            const answers = [q1, q2, q3, q4, q5];

            let teacherCount = 0;

            let trainerCount = 0;

            let tutorCount = 0;

           

            answers.forEach(answer => {

                if (answer === 'A') teacherCount++;

                if (answer === 'B') trainerCount++;

                if (answer === 'C') tutorCount++;

            });

           

            // Calculate percentages

            const total = 5; // total number of questions

            const teacherPercent = (teacherCount / total) * 100;

            const trainerPercent = (trainerCount / total) * 100;

            const tutorPercent = (tutorCount / total) * 100;

           

            // Update progress bars

            document.getElementById('teacher-score-bar').style.width = teacherPercent + '%';

            document.getElementById('trainer-score-bar').style.width = trainerPercent + '%';

            document.getElementById('tutor-score-bar').style.width = tutorPercent + '%';

           

            // Update percentage values

            document.getElementById('teacher-score-value').textContent = teacherPercent + '%';

            document.getElementById('trainer-score-value').textContent = trainerPercent + '%';

            document.getElementById('tutor-score-value').textContent = tutorPercent + '%';

           

            // Determine primary path

            let primaryPath = '';

            let pathDescription = '';

           

            if (teacherCount >= trainerCount && teacherCount >= tutorCount) {

                primaryPath = 'Teacher Path';

                pathDescription = `

                    <div class="path-header">

                        <div class="path-icon teacher-icon">T</div>

                        <h3>Teacher Path</h3>

                    </div>

                    <p><strong>Focus:</strong> Building strong independent skills first, then using AI as feedback and enhancement.</p>

                    <p>This path is designed for learners who need to develop core competencies without AI assistance. You'll benefit from focusing on building independent skills first, then gradually incorporating AI as a feedback mechanism.</p>

                    <p>The Teacher path emphasizes developing critical skills before incorporating AI tools and addresses concerns about overreliance on AI. It gradually introduces AI as a review mechanism and learning support.</p>

                `;

            } else if (trainerCount >= teacherCount && trainerCount >= tutorCount) {

                primaryPath = 'Trainer Path';

                pathDescription = `

                    <div class="path-header">

                        <div class="path-icon trainer-icon">T</div>

                        <h3>Trainer Path</h3>

                    </div>

                    <p><strong>Focus:</strong> Applying AI to technical and vocational applications.</p>

                    <p>This path is designed for those applying AI to specific workplace or technical challenges. This approach helps you apply AI to technical and workplace challenges with an emphasis on practical applications.</p>

                    <p>The Trainer path emphasizes practical applications and industry-specific uses, focuses on prompting techniques for technical problem-solving, and helps integrate AI into existing technical workflows.</p>

                `;

            } else {

                primaryPath = 'Tutor Path';

                pathDescription = `

                    <div class="path-header">

                        <div class="path-icon tutor-icon">T</div>

                        <h3>Tutor Path</h3>

                    </div>

                    <p><strong>Focus:</strong> Enhancing productivity and creativity with AI partnership.</p>

                    <p>This path is designed for advanced learners who already possess strong foundational skills. This approach leverages your existing expertise while using AI as a collaborative partner to enhance your productivity and creativity.</p>

                    <p>The Tutor path uses AI as a collaborative thought partner to extend capabilities, explores advanced techniques for complex applications, and emphasizes leveraging AI to overcome creative blocks and enhance existing strengths.</p>

                `;

            }

           

            document.getElementById('primary-path').innerHTML = primaryPath;

            document.getElementById('path-description').innerHTML = pathDescription;

           

            // Show mixed message if scores are distributed

            if (teacherCount === trainerCount || teacherCount === tutorCount || trainerCount === tutorCount) {

                document.getElementById('mixed-message').style.display = 'block';

            } else {

                document.getElementById('mixed-message').style.display = 'none';

            }

           

            // Show results

            document.getElementById('result').style.display = 'block';

           

            // Scroll to results

            document.getElementById('result').scrollIntoView({ behavior: 'smooth' });

        });
